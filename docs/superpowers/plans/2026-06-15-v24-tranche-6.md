# v24 — Tranche 6: Track Issue Row Redesign

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the DataTable-based issue list in Track with a custom dense row list matching `linear0.png`. Each row: colored status circle (CSS `data-status` attribute), issue ID (`TRK-{n}`), issue title, spacer, assignee `Avatar` (sm), relative date. Row height ≤36px. Above the rows: a content-area header with `Breadcrumb` ("Forum / Issues"), issue count `Badge`, and placeholder filter/sort/view icon buttons. The `issueTab` signal (All / Active / Backlog) from T5 filters visible rows.

**Architecture:** New `IssueRow.tsx` is a pure display component. New `IssueListView.tsx` wraps Tabs + issue list + header. It reads `issueTab` from the app-level signal and filters `issues` signal from the store. Replaces the `<List />` + `<Tabs>` block in `App.tsx`.

**Tech Stack:** `@cascivo/react` (Avatar, Badge, Tabs, TabsList, TabsTrigger, TabsContent, Breadcrumb, Tooltip), `@cascivo/i18n`, `@cascivo/core` (useSignals, useComputed).

- [ ] **Step 3: Create IssueRow.tsx**

  ```tsx
  'use client'
  import { useSignals } from '@cascivo/core'
  import { Avatar, Tooltip } from '@cascivo/react'
  import type { Issue } from '../store/issues'
  import { USERS } from '../data/seed'
  import { editingIssueId } from '../commands'
  import styles from './IssueRow.module.css'

  interface Props {
    issue: Issue
    index: number // 1-based for TRK-{n}
  }

  function relativeDate(iso: string): string {
    const diff = Date.now() - new Date(iso).getTime()
    const days = Math.floor(diff / 86_400_000)
    if (days === 0) return 'today'
    if (days === 1) return 'yesterday'
    if (days < 30) return `${days}d ago`
    return `${Math.floor(days / 30)}mo ago`
  }

  export function IssueRow({ issue, index }: Props) {
    useSignals()

    const assignee = USERS.find((u) => u.id === issue.assigneeId)

    return (
      <button
        type="button"
        className={styles['row']}
        onClick={() => {
          editingIssueId.value = issue.id
        }}
      >
        {/* Status circle */}
        <span className={styles['statusCircle']} data-status={issue.status} aria-hidden />

        {/* ID */}
        <span className={styles['issueId']}>TRK-{index}</span>

        {/* Title */}
        <span className={styles['title']}>{issue.title}</span>

        {/* Spacer */}
        <span className={styles['spacer']} />

        {/* Assignee avatar */}
        {assignee ? (
          <Tooltip content={assignee.name}>
            <Avatar name={assignee.name} size="sm" />
          </Tooltip>
        ) : (
          <span className={styles['avatarPlaceholder']} />
        )}

        {/* Date */}
        <span className={styles['date']}>{relativeDate(issue.updatedAt)}</span>
      </button>
    )
  }
  ```

- [ ] **Step 3: Create IssueListView.tsx**

  ```tsx
  'use client'
  import { useSignals, useComputed } from '@cascivo/core'
  import { t } from '@cascivo/i18n'
  import {
    Breadcrumb,
    Badge,
    Tabs,
    TabsList,
    TabsTrigger,
    TabsContent,
    Tooltip,
  } from '@cascivo/react'
  import { issues } from '../store/issues'
  import { issueTab, assigneeFilter } from '../App'
  import { IssueRow } from './IssueRow'
  import { msg } from '../i18n'
  import styles from './IssueListView.module.css'

  const ACTIVE_STATUSES = new Set(['todo', 'in-progress', 'in-review'])
  const BACKLOG_STATUSES = new Set(['backlog'])

  export function IssueListView() {
    useSignals()

    const filtered = useComputed(() => {
      const all = issues.value
      const filter = assigneeFilter.value
      const tab = issueTab.value
      return all
        .filter((i) => (filter ? i.assigneeId === filter : true))
        .filter((i) => {
          if (tab === 'active') return ACTIVE_STATUSES.has(i.status)
          if (tab === 'backlog') return BACKLOG_STATUSES.has(i.status)
          return true
        })
    })

    const teamName = t(msg.navTeamName)
    const issueCount = filtered.value.length

    return (
      <div className={styles['root']}>
        {/* Content header */}
        <header className={styles['header']}>
          {/* Breadcrumb uses items-array API — no compound sub-components */}
          <div className={styles['breadcrumbWrap']}>
            <Breadcrumb items={[{ label: teamName }, { label: t(msg.navSubIssues) }]} />
            <Badge size="sm" variant="secondary" className={styles['countBadge']}>
              {issueCount}
            </Badge>
          </div>
          <div className={styles['headerActions']}>
            <Tooltip content={t(msg.toolbarFilter)}>
              <button type="button" className={styles['iconBtn']} aria-label={t(msg.toolbarFilter)}>
                ⊟
              </button>
            </Tooltip>
            <Tooltip content={t(msg.toolbarSort)}>
              <button type="button" className={styles['iconBtn']} aria-label={t(msg.toolbarSort)}>
                ↕
              </button>
            </Tooltip>
          </div>
        </header>

        {/* Sub-tabs */}
        <Tabs
          value={issueTab.value}
          onValueChange={(v) => {
            issueTab.value = v as typeof issueTab.value
          }}
        >
          <TabsList>
            <TabsTrigger value="all">{t(msg.issueTabAll)}</TabsTrigger>
            <TabsTrigger value="active">{t(msg.issueTabActive)}</TabsTrigger>
            <TabsTrigger value="backlog">{t(msg.issueTabBacklog)}</TabsTrigger>
          </TabsList>

          {(['all', 'active', 'backlog'] as const).map((tab) => (
            <TabsContent key={tab} value={tab}>
              {/* Team section header */}
              <div className={styles['teamHeader']}>{teamName}</div>
              <div className={styles['issueList']}>
                {filtered.value.length === 0 ? (
                  <p className={styles['empty']}>{t(msg.listEmpty)}</p>
                ) : (
                  filtered.value.map((issue, i) => (
                    <IssueRow key={issue.id} issue={issue} index={i + 1} />
                  ))
                )}
              </div>
            </TabsContent>
          ))}
        </Tabs>
      </div>
    )
  }
  ```

```tsx
'use client'

import { useRef } from 'react'
import { useSignal, useSignalEffect } from '@cascade-ui/core'
import styles from './streaming-text.module.css'

export interface StreamingTextProps {
  text: string
  /** Characters to advance per animation frame. Default: 2 */
  speed?: number
  onComplete?: () => void
  className?: string
}

export function StreamingText({ text, speed = 2, onComplete, className }: StreamingTextProps) {
  const displayed = useSignal('')
  const rafRef = useRef<number | undefined>(undefined)
  const onCompleteRef = useRef(onComplete)
  onCompleteRef.current = onComplete

  useSignalEffect(() => {
    // Respect reduced-motion preference: render final state immediately
    const prefersReduced =
      typeof window !== 'undefined' && window.matchMedia('(prefers-reduced-motion: reduce)').matches

    if (prefersReduced) {
      displayed.value = text
      onCompleteRef.current?.()
      return
    }

    displayed.value = ''
    let pos = 0

    function tick() {
      pos = Math.min(pos + speed, text.length)
      displayed.value = text.slice(0, pos)
      if (pos < text.length) {
        rafRef.current = requestAnimationFrame(tick)
      } else {
        onCompleteRef.current?.()
      }
    }

    rafRef.current = requestAnimationFrame(tick)

    return () => {
      if (rafRef.current !== undefined) cancelAnimationFrame(rafRef.current)
    }
  })

  return (
    <span
      role="status"
      aria-live="polite"
      aria-atomic="false"
      data-streaming={displayed.value.length < text.length ? 'true' : 'false'}
      className={[styles.root, className].filter(Boolean).join(' ')}
    >
      {displayed.value}
    </span>
  )
}
```

```tsx
'use client'

import { useRef } from 'react'
import { useSignal, useSignalEffect } from '@cascade-ui/core'
import styles from './terminal.module.css'

export type TerminalLineType = 'command' | 'output' | 'error' | 'comment'

export interface TerminalLine {
  text: string
  prefix?: string
  type?: TerminalLineType
  delayMs?: number
  speed?: number
}

export interface TerminalProps {
  lines: TerminalLine[]
  speed?: number
  loop?: boolean
  onComplete?: () => void
  className?: string
}

interface DisplayLine {
  text: string
  prefix?: string
  type?: TerminalLineType
  complete: boolean
}

export function Terminal({ lines, speed = 3, loop, onComplete, className }: TerminalProps) {
  const displayed = useSignal<DisplayLine[]>([])
  const rafRef = useRef<number | undefined>(undefined)
  const onCompleteRef = useRef(onComplete)
  onCompleteRef.current = onComplete

  useSignalEffect(() => {
    const prefersReduced =
      typeof window !== 'undefined' && window.matchMedia('(prefers-reduced-motion: reduce)').matches

    if (prefersReduced) {
      displayed.value = lines.map((l) => ({ ...l, complete: true }))
      onCompleteRef.current?.()
      return
    }

    displayed.value = []
    let lineIdx = 0
    let charIdx = 0

    function tick() {
      const line = lines[lineIdx]
      if (!line) {
        onCompleteRef.current?.()
        if (loop) {
          lineIdx = 0
          charIdx = 0
          displayed.value = []
          rafRef.current = requestAnimationFrame(tick)
        }
        return
      }

      const lineSpeed = line.speed ?? speed
      charIdx = Math.min(charIdx + lineSpeed, line.text.length)

      const current = displayed.value
      const existingLine = current[lineIdx]
      const updatedLine: DisplayLine = {
        text: line.text.slice(0, charIdx),
        prefix: line.prefix,
        type: line.type,
        complete: charIdx >= line.text.length,
      }

      displayed.value = [...current.slice(0, lineIdx), updatedLine]

      if (charIdx >= line.text.length) {
        lineIdx++
        charIdx = 0
        // Respect per-line delay before starting next line
        const nextDelay = lines[lineIdx]?.delayMs ?? 0
        if (nextDelay > 0) {
          setTimeout(() => {
            rafRef.current = requestAnimationFrame(tick)
          }, nextDelay)
          return
        }
      }

      rafRef.current = requestAnimationFrame(tick)
    }

    rafRef.current = requestAnimationFrame(tick)

    return () => {
      if (rafRef.current !== undefined) cancelAnimationFrame(rafRef.current)
    }
  })

  return (
    <div
      role="log"
      aria-live="polite"
      aria-label="Terminal output"
      data-component="terminal"
      className={[styles.root, className].filter(Boolean).join(' ')}
    >
      {displayed.value.map((line, i) => (
        <div
          key={i}
          data-type={line.type ?? 'output'}
          data-complete={line.complete}
          className={styles.line}
        >
          {line.prefix && (
            <span className={styles.prefix} aria-hidden="true">
              {line.prefix}
            </span>
          )}
          <span className={styles.text}>{line.text}</span>
          {!line.complete && (
            <span className={styles.cursor} aria-hidden="true">
              ▋
            </span>
          )}
        </div>
      ))}
    </div>
  )
}
```

`packages/ai/src/markdown.ts` — hand-rolled, no remark dependency (bundle budget: < 20 KB):

````ts
/**
 * Minimal markdown renderer — bold, italic, inline code, fenced code blocks,
 * links, and unordered/ordered lists. No full CommonMark — intentional scope limit.
 * Output: sanitized HTML string (XSS-safe: only allowlisted tags).
 */
export function renderMarkdown(text: string): string {
  let html = text
    // Escape HTML entities first
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')

  // Fenced code blocks (must be before inline code)
  html = html.replace(/```(\w*)\n([\s\S]*?)```/g, (_, lang, code) => {
    const attr = lang ? ` data-language="${lang}"` : ''
    return `<pre${attr}><code>${code.trimEnd()}</code></pre>`
  })

  // Inline code
  html = html.replace(/`([^`]+)`/g, '<code>$1</code>')

  // Bold
  html = html.replace(/\*\*([^*]+)\*\*/g, '<strong>$1</strong>')

  // Italic
  html = html.replace(/\*([^*]+)\*/g, '<em>$1</em>')

  // Links
  html = html.replace(/\[([^\]]+)\]\(([^)]+)\)/g, '<a href="$2" rel="noopener noreferrer">$1</a>')

  // Unordered lists
  html = html.replace(/^[*-] (.+)$/gm, '<li>$1</li>')
  html = html.replace(/(<li>.*<\/li>\n?)+/g, '<ul>$&</ul>')

  // Ordered lists
  html = html.replace(/^\d+\. (.+)$/gm, '<li>$1</li>')

  // Paragraphs (double newline)
  html = html.replace(/\n{2,}/g, '</p><p>')
  html = `<p>${html}</p>`

  // Clean up empty paragraphs around block elements
  html = html.replace(/<p>(<pre|<ul|<ol)/g, '$1')
  html = html.replace(/(<\/pre>|<\/ul>|<\/ol>)<\/p>/g, '$1')

  return html
}
````

```tsx
'use client'

import { useRef, type KeyboardEvent } from 'react'
import { useSignal, useSignalEffect, useSignals } from '@cascade-ui/core'
import { builtin, t } from '@cascade-ui/i18n'
import { StreamingText } from './streaming-text'
import { renderMarkdown } from './markdown'
import styles from './ai-chat.module.css'

export type MessageRole = 'user' | 'assistant' | 'system'

export interface ChatMessage {
  id: string
  role: MessageRole
  content: string
}

export interface AiChatProps {
  messages: ChatMessage[]
  onSend: (text: string) => void
  isStreaming?: boolean
  streamingText?: string
  labels?: {
    send?: string
    placeholder?: string
    you?: string
    assistant?: string
  }
}

export function AiChat({ messages, onSend, isStreaming, streamingText, labels }: AiChatProps) {
  useSignals()
  const inputValue = useSignal('')
  const scrollRef = useRef<HTMLDivElement>(null)
  const onSendRef = useRef(onSend)
  onSendRef.current = onSend

  // Auto-scroll to bottom when messages change, unless user has scrolled up
  const userScrolledUp = useSignal(false)

  useSignalEffect(() => {
    void messages.length // subscribe to message changes
    if (!userScrolledUp.value) {
      scrollRef.current?.scrollTo({ top: scrollRef.current.scrollHeight, behavior: 'smooth' })
    }
  })

  function handleSend() {
    const text = inputValue.value.trim()
    if (!text) return
    inputValue.value = ''
    userScrolledUp.value = false
    onSendRef.current(text)
  }

  function handleKeyDown(e: KeyboardEvent<HTMLTextAreaElement>) {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault()
      handleSend()
    }
  }

  function handleScroll() {
    const el = scrollRef.current
    if (!el) return
    const atBottom = el.scrollHeight - el.scrollTop - el.clientHeight < 40
    userScrolledUp.value = !atBottom
  }

  return (
    <div data-component="ai-chat" className={styles.root}>
      <div
        ref={scrollRef}
        role="log"
        aria-live="polite"
        className={styles.messages}
        onScroll={handleScroll}
      >
        {messages.map((msg) => (
          <div key={msg.id} data-role={msg.role} className={styles.message}>
            <span className={styles.roleLabel}>
              {msg.role === 'user'
                ? (labels?.you ?? t(builtin.ai.you))
                : (labels?.assistant ?? t(builtin.ai.assistant))}
            </span>
            {msg.role === 'assistant' ? (
              <div
                className={styles.content}
                // eslint-disable-next-line react/no-danger
                dangerouslySetInnerHTML={{ __html: renderMarkdown(msg.content) }}
              />
            ) : (
              <div className={styles.content}>{msg.content}</div>
            )}
          </div>
        ))}
        {isStreaming && streamingText !== undefined && (
          <div data-role="assistant" className={styles.message}>
            <span className={styles.roleLabel}>{labels?.assistant ?? t(builtin.ai.assistant)}</span>
            <div className={styles.content}>
              <StreamingText text={streamingText} />
            </div>
          </div>
        )}
      </div>
      <div className={styles.inputArea}>
        <textarea
          aria-label={labels?.placeholder ?? t(builtin.ai.placeholder)}
          placeholder={labels?.placeholder ?? t(builtin.ai.placeholder)}
          rows={1}
          value={inputValue.value}
          onChange={(e) => {
            inputValue.value = e.target.value
          }}
          onKeyDown={handleKeyDown}
          className={styles.input}
        />
        <button
          type="button"
          aria-label={labels?.send ?? t(builtin.ai.send)}
          onClick={handleSend}
          disabled={!inputValue.value.trim()}
          className={styles.sendButton}
        >
          {labels?.send ?? t(builtin.ai.send)}
        </button>
      </div>
    </div>
  )
}
```

```tsx
import { useSignal, useSignals } from '@cascade-ui/core'
import { AiChat, type ChatMessage } from '@cascade-ui/ai'

let msgId = 0

export function AiChatPage() {
  useSignals()
  const messages = useSignal<ChatMessage[]>([])
  const isStreaming = useSignal(false)
  const streamingText = useSignal('')

  function handleSend(text: string) {
    messages.value = [...messages.value, { id: String(++msgId), role: 'user', content: text }]
    isStreaming.value = true
    streamingText.value = ''

    // Demo: echo with a simulated streaming response
    const response = `You said: "${text}" — this is a demo response from the AI assistant.`
    let i = 0
    const interval = setInterval(() => {
      streamingText.value = response.slice(0, ++i)
      if (i >= response.length) {
        clearInterval(interval)
        messages.value = [
          ...messages.value,
          { id: String(++msgId), role: 'assistant', content: response },
        ]
        isStreaming.value = false
        streamingText.value = ''
      }
    }, 20)
  }

  return (
    <div style={{ height: '100vh', display: 'flex', flexDirection: 'column' }}>
      <AiChat
        messages={messages.value}
        onSend={handleSend}
        isStreaming={isStreaming.value}
        streamingText={streamingText.value}
      />
    </div>
  )
}
```

```ts
import { test, expect } from '@playwright/test'

test('AiChat streaming: 60fps with zero React commits per token', async ({ page }) => {
  await page.goto('/iframe.html?id=ai-aichat--default')

  // Start streaming via the send action
  const input = page.locator('textarea')
  await input.fill('benchmark test')
  await input.press('Enter')

  // Measure frame timing during streaming (must stay >= 55fps = ~18ms/frame)
  const timings = await page.evaluate(async () => {
    return new Promise<number[]>((resolve) => {
      const frames: number[] = []
      let last = performance.now()
      function measure() {
        const now = performance.now()
        frames.push(now - last)
        last = now
        if (frames.length < 60) requestAnimationFrame(measure)
        else resolve(frames)
      }
      requestAnimationFrame(measure)
    })
  })

  const avgFrameTime = timings.reduce((a, b) => a + b, 0) / timings.length
  // 60fps = 16.67ms per frame; allow up to 20ms average (55fps)
  expect(avgFrameTime).toBeLessThan(20)
})
```

**Goal:** Implement `app-shell` (sidebar layout with signal-driven mobile toggle) and
`settings-profile` (two-section settings form). Extend `registry.json` with a `blocks` array.
Update the CLI registry resolver to handle the `block/` prefix. Extend the screenshot pipeline to
capture all 8 blocks.

**Architecture:** `app-shell` uses `useSignal(false)` for sidebar open/close — the only block
with FSM-like internal state. `settings-profile` is purely form state (signals for each field).
The registry extension adds a `blocks` top-level key to `registry.json`; the CLI reads from it
when the name starts with `block/`. Screenshots use the existing Playwright-based pipeline.

**Tech Stack:** `@cascivo/core` (useSignal, useSignals, useSignalEffect), `@cascivo/react`
(Button, Input, Textarea, Select, Toggle, Card), Node.js (registry script), Playwright
(screenshots), Vitest + @testing-library/react.

- [ ] **Step 2: Create block component**

  Create `packages/components/src/blocks/app-shell/app-shell.tsx`:

  ```tsx
  import { useSignal, useSignalEffect, useSignals } from '@cascivo/core'
  import { Button } from '@cascivo/react/button'
  import styles from './app-shell.module.css'

  type NavItem = { label: string; href: string; icon: string; current?: boolean }

  const NAV_ITEMS: NavItem[] = [
    { label: 'Dashboard', href: '#', icon: '🏠', current: true },
    { label: 'Analytics', href: '#', icon: '📊' },
    { label: 'Customers', href: '#', icon: '👥' },
    { label: 'Orders', href: '#', icon: '📦' },
    { label: 'Settings', href: '#', icon: '⚙️' },
  ]

  export function AppShell() {
    useSignals()
    const sidebarOpen = useSignal(false)

    useSignalEffect(() => {
      document.body.style.overflow = sidebarOpen.value ? 'hidden' : ''
      return () => {
        document.body.style.overflow = ''
      }
    })

    function toggleSidebar() {
      sidebarOpen.value = !sidebarOpen.value
    }

    function closeSidebar() {
      sidebarOpen.value = false
    }

    return (
      <div className={styles.root}>
        {/* Mobile top bar */}
        <div className={styles.topbar}>
          <span className={styles.topbarLogo}>cascivo</span>
          <Button
            variant="ghost"
            size="sm"
            aria-label="Open menu"
            aria-expanded={sidebarOpen.value}
            onClick={toggleSidebar}
          >
            ☰
          </Button>
        </div>

        {/* Sidebar overlay (mobile) */}
        {sidebarOpen.value && (
          <div className={styles.overlay} onClick={closeSidebar} aria-hidden="true" />
        )}

        {/* Sidebar */}
        <aside className={styles.sidebar} data-open={sidebarOpen.value}>
          <div className={styles.sidebarLogo}>cascivo</div>
          <nav className={styles.nav} aria-label="Main navigation">
            {NAV_ITEMS.map((item) => (
              <a
                key={item.label}
                href={item.href}
                className={styles.navLink}
                aria-current={item.current ? 'page' : undefined}
              >
                <span aria-hidden="true">{item.icon}</span>
                {item.label}
              </a>
            ))}
          </nav>
        </aside>

        {/* Main content */}
        <main className={styles.content}>
          <p className={styles.breadcrumb}>Home / Dashboard</p>
          <h1 className={styles.pageTitle}>Dashboard</h1>
          <div className={styles.contentPlaceholder}>Page content goes here</div>
        </main>
      </div>
    )
  }
  ```

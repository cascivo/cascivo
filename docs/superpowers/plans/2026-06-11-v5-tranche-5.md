- [ ] **Step 2: Rewrite Layout.** Target shape (adapt prop names to what Step 1 found — this is

```tsx
import { useMemo } from 'preact/hooks'
import { ShellHeader } from '@cascivo/components/shell-header'
import { SideNav } from '@cascivo/components/side-nav'
import { HeaderPanel } from '@cascivo/components/header-panel'
import { AppShell } from '@cascivo/layouts/app-shell'
import { createShellState } from '@cascivo/layouts/shell-state'
import { useSignal } from '@preact/signals'
import { buildNav } from './nav'
import { theme, setTheme, THEMES } from './theme' // wherever App.tsx keeps it — extract if inline

const shell = createShellState({ persistKey: 'docs.shell' }) // module scope — page-wide singleton

export function Layout({ children, path }: { children: preact.ComponentChildren; path: string }) {
  const themePanelOpen = useSignal(false)
  const nav = useMemo(buildNav, [])

  const sideNavItems = [
    {
      label: 'Explore',
      items: [
        { label: 'AI / MCP', href: '/ai', active: path === '/ai' },
        { label: 'Charts', href: '/charts', active: path === '/charts' },
      ],
    },
    ...nav.map((group) => ({
      label: group.label,
      items: group.items.map((item) => ({
        label: item.label,
        href: item.href,
        active: path === item.href,
      })),
    })),
  ]

  return (
    <AppShell
      state={shell}
      persistKey="docs.shell"
      header={
        <ShellHeader
          brand={{ prefix: 'cascade', name: 'Docs', href: '/' }}
          onMenuClick={shell.toggleSideNav}
          menuExpanded={!shell.sideNavCollapsed.value}
          actions={[
            {
              id: 'theme',
              label: 'Theme',
              icon: <PaletteIcon />, // use an icon from @cascivo/icons (e.g. Settings or Sun/Moon)
              active: themePanelOpen.value,
              onClick: () => {
                themePanelOpen.value = !themePanelOpen.value
              },
            },
          ]}
        />
      }
      sideNav={
        <SideNav
          items={sideNavItems}
          collapsed={shell.sideNavCollapsed.value}
          onCollapsedChange={(c) => {
            shell.sideNavCollapsed.value = c
          }}
        />
      }
    >
      <HeaderPanel
        open={themePanelOpen.value}
        onClose={() => {
          themePanelOpen.value = false
        }}
        label="Theme"
      >
        {/* 5 theme rows: button per theme, aria-pressed for the active one, calls setTheme */}
      </HeaderPanel>
      <div class="page">{children}</div>
    </AppShell>
  )
}
```

Implementation notes:

- Docs is an APP, not a library component — Preact hooks/signals are fine here; CLAUDE.md's
  hook ban applies to `packages/components`, not apps.
- `createShellState` at module scope is the documented usage (v4 master plan decision); the
  persisted keys (`docs.shell.sidenav`, `docs.shell.aside`) replace the old
  `persistedSignal('docs.sidebar-collapsed')` — drop the old key.
- SideNav groups: nav groups become group items (`{ label, items }` without href) — verify
  SideNav renders non-collapsed groups expanded by default with the active leaf highlighted;
  if groups default collapsed, set whatever `defaultExpanded`/open prop exists, or pass active
  state so the group containing the route opens (read side-nav.tsx behavior and use it).
- Theme switching: keep writing `data-theme` to `document.documentElement` (the existing
  `App.tsx` mechanism) — only the UI moves into the HeaderPanel.
- Keep the skip-link behavior: AppShell renders `main#cascade-main` and ShellHeader's skip link
  targets it — free.

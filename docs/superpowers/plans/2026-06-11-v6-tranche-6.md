- [ ] **Step 2: Implement** `src/report.ts`. Required structure of the emitted markdown (write
      the renderer to produce exactly these sections, each conditional on its results.json
      section existing):

````ts
import type { Results } from './types.ts'

const LIBS = ['cascade', 'shadcn', 'carbon'] as const
const ms = (v: number) => `${v.toFixed(1)}ms`

export function renderReport(r: Results): string {
  const parts: string[] = []

  parts.push(
    '# cascade benchmarks',
    '',
    '> Generated by `pnpm bench` — do not edit. Methodology: `apps/bench/METHODOLOGY.md`.',
    '> Reproduce: clone, `pnpm install`, `pnpm bench`.',
    '',
    '## Environment',
    '',
    `| | |`,
    `| --- | --- |`,
    `| Date | ${r.meta.date} |`,
    `| CPU | ${r.meta.cpu} (${r.meta.cores} cores, ${r.meta.memGb}GB) |`,
    `| OS / Node / Chrome | ${r.meta.os} / ${r.meta.node} / ${r.meta.chrome} |`,
    `| CPU throttle (timed ops) | ${r.meta.cpuThrottle}× |`,
    `| Lockfile | ${r.meta.lockfileHash} |`,
    `| Source | ${r.meta.source} |`,
    '',
  )

  if (r.bundle) {
    parts.push('## Bundle size (min+gzip, level 6)', '')
    parts.push('| App | JS | CSS | Total |', '| --- | ---: | ---: | ---: |')
    for (const lib of LIBS) {
      const a = r.bundle.apps[lib]
      if (a) parts.push(`| ${lib} | ${a.jsGzKb}KB | ${a.cssGzKb}KB | **${a.totalGzKb}KB** |`)
    }
    // … incremental matrix table (component rows × library columns, incrementalGzKb)
    // … treeshake block if present
    parts.push('')
  }

  if (r.runtime) {
    parts.push(
      '## Interaction latency (Chrome trace, click→paint)',
      '',
      `Median of ≥12 samples at ${r.meta.cpuThrottle}× CPU throttle; IQR in parentheses.`,
      'Deltas vs cascade with Mann-Whitney p ≥ 0.05 are reported as **tie** — no winner is claimed.',
      '',
      '| Scenario | cascade | shadcn | carbon |',
      '| --- | ---: | ---: | ---: |',
    )
    for (const [id, row] of Object.entries(r.runtime)) {
      const cells = LIBS.map((lib) => {
        const s = row[lib]
        if (!s) return 'n/a'
        const p = lib === 'cascade' ? undefined : row.pVsCascade?.[lib]
        const tie = p !== undefined && p >= 0.05 ? ' (tie)' : ''
        return `${ms(s.median)} (${ms(s.p25)}–${ms(s.p75)})${tie}`
      })
      parts.push(`| ${id} | ${cells.join(' | ')} |`)
    }
    parts.push('')
  }

  if (r.renders) {
    parts.push(
      '## Re-render counts (React Profiler root commits)',
      '',
      'Deterministic integers from instrumented dev builds (Profiler is a production no-op).',
      'Timings above are NOT from these builds.',
      '',
      '| Scenario | cascade | shadcn | carbon |',
      '| --- | ---: | ---: | ---: |',
    )
    for (const [id, row] of Object.entries(r.renders)) {
      parts.push(`| ${id} | ${LIBS.map((l) => row[l] ?? 'n/a').join(' | ')} |`)
    }
    parts.push('')
  }

  if (r.lighthouse) {
    parts.push(
      '## Lighthouse (median of 5 runs, desktop preset)',
      '',
      'TBT is a lab proxy for INP — INP itself cannot be measured in a lab.',
      '',
      '| App | FCP | LCP | TBT | Transfer |',
      '| --- | ---: | ---: | ---: | ---: |',
    )
    for (const lib of LIBS) {
      const l = r.lighthouse[lib]
      if (l) parts.push(`| ${lib} | ${l.fcpMs}ms | ${l.lcpMs}ms | ${l.tbtMs}ms | ${l.transferKb}KB |`)
    }
    parts.push('')
  }

  if (r.a11y) {
    parts.push(
      '## Accessibility (axe-core sweep — parity gate, not a score)',
      '',
      'Automated tools detect roughly 57% of WCAG issues at best; these numbers are a floor,',
      'not a ranking. cascade CI fails on any violation. Competitor numbers are context only.',
      '',
      '| App | Violations | Rules |',
      '| --- | ---: | --- |',
    )
    for (const lib of LIBS) {
      const a = r.a11y[lib]
      if (a) parts.push(`| ${lib} | ${a.violations} | ${a.rules.join(', ') || '—'} |`)
    }
    parts.push('')
  }

  return `${parts.join('\n')}\n`
}
````

(The two `// …` lines mark tables to write out fully in the same style as the surrounding

`bareImportGzBytes`, `buttonOnlyGzKb`, `fullGzKb`. No conditional formatting by winner

5. **Lighthouse**: median-of-5, desktop preset, TBT-as-INP-proxy disclaimer.
6. **a11y**: parity-gate framing, axe ~57% detection ceiling citation, the 4-state matrix.
7. **What we don't measure (yet)**: SSR/hydration, field data (real INP), MUI/Mantine — with

- [ ] **Step 2:** Create `apps/bench/runner/src/ci-compare.ts` — compares the CI-produced
      deterministic sections against the committed ones from git (`git show
HEAD:apps/bench/results/results.json`): render counts must be identical; bundle app totals
      within ±2%; exits non-zero with a readable diff otherwise:

```ts
import { execSync } from 'node:child_process'
import { readFileSync } from 'node:fs'
import type { Results } from './types.ts'

const repoRoot = new URL('../../../..', import.meta.url).pathname
const current = JSON.parse(
  readFileSync(`${repoRoot}/apps/bench/results/results.json`, 'utf8'),
) as Results
const committed = JSON.parse(
  execSync('git show HEAD:apps/bench/results/results.json', { cwd: repoRoot }).toString(),
) as Results

const failures: string[] = []

for (const [scenario, row] of Object.entries(committed.renders ?? {})) {
  for (const [lib, count] of Object.entries(row)) {
    const now = current.renders?.[scenario as keyof typeof current.renders]?.[lib as 'cascade']
    if (now !== count) failures.push(`renders ${scenario}/${lib}: committed ${count}, CI ${now}`)
  }
}

for (const [lib, app] of Object.entries(committed.bundle?.apps ?? {})) {
  const now = current.bundle?.apps[lib as 'cascade']
  if (!now || Math.abs(now.totalGzKb - app.totalGzKb) / app.totalGzKb > 0.02) {
    failures.push(`bundle ${lib}: committed ${app.totalGzKb}KB, CI ${now?.totalGzKb}KB (>2%)`)
  }
}

if (failures.length > 0) {
  console.error(failures.join('\n'))
  process.exit(1)
}
console.log('✓ CI results match committed results within tolerance')
```

Important: CI must NOT overwrite the committed timing sections — `runtime`/`lighthouse` runs in
CI write to the working copy only (artifact), never committed. The `deterministic` job runs
renders AFTER bundle so the artifact carries both; nothing in CI commits.

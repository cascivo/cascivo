### Task 3: Bundle budget gate

**Files:**

- Create: `scripts/quality/bundle-check.ts`
- Modify: `.github/workflows/ci.yml`

- [ ] **Step 1: Write the bundle check script**

```ts
import { execSync } from 'node:child_process'
import { statSync } from 'node:fs'
import { join } from 'node:path'
import { fileURLToPath } from 'node:url'
import { createGzip } from 'node:zlib'
import { createReadStream } from 'node:fs'

const root = fileURLToPath(new URL('../..', import.meta.url))

const BUDGETS: { pkg: string; distFile: string; maxGzipKB: number }[] = [
  { pkg: '@cascade-ui/react', distFile: 'packages/react/dist/index.mjs', maxGzipKB: 50 },
  { pkg: '@cascade-ui/ai', distFile: 'packages/ai/dist/index.mjs', maxGzipKB: 20 },
  { pkg: '@cascade-ui/charts', distFile: 'packages/charts/dist/index.mjs', maxGzipKB: 80 },
]

async function gzipSize(filePath: string): Promise<number> {
  return new Promise((resolve, reject) => {
    let size = 0
    const gzip = createGzip()
    const stream = createReadStream(filePath)
    gzip.on('data', (chunk) => {
      size += (chunk as Buffer).length
    })
    gzip.on('end', () => resolve(size))
    gzip.on('error', reject)
    stream.on('error', reject)
    stream.pipe(gzip)
  })
}

const failures: string[] = []

for (const { pkg, distFile, maxGzipKB } of BUDGETS) {
  const fullPath = join(root, distFile)
  try {
    const bytes = await gzipSize(fullPath)
    const kb = bytes / 1024
    if (kb > maxGzipKB) {
      failures.push(`${pkg}: ${kb.toFixed(1)} KB gzip > budget ${maxGzipKB} KB`)
    } else {
      console.log(`✓ ${pkg}: ${kb.toFixed(1)} KB gzip (budget: ${maxGzipKB} KB)`)
    }
  } catch (e) {
    failures.push(`${pkg}: could not read dist file ${distFile} — ${String(e)}`)
  }
}

if (failures.length > 0) {
  console.error('\nBundle budget failures:\n' + failures.map((f) => `  ✗ ${f}`).join('\n'))
  process.exit(1)
}
console.log('\nAll bundle budgets passed.')
```

### Task 9: Storybook completeness check

**Files:**

- Create: `scripts/quality/story-check.ts`
- Modify: `.github/workflows/ci.yml`

- [ ] **Step 1: Write the completeness check**

```ts
import { readdirSync, existsSync } from 'node:fs'
import { join } from 'node:path'
import { fileURLToPath } from 'node:url'

const root = fileURLToPath(new URL('../..', import.meta.url))
const registry = JSON.parse(require('fs').readFileSync(join(root, 'registry.json'), 'utf8')) as {
  components: { name: string }[]
}

const storiesDir = join(root, 'apps', 'storybook', 'stories')

const missing: string[] = []
for (const { name } of registry.components) {
  const kebab = name.replace(/([a-z])([A-Z])/g, '$1-$2').toLowerCase()
  const story = join(storiesDir, `${kebab}.stories.tsx`)
  if (!existsSync(story)) {
    missing.push(`${name}: missing story at stories/${kebab}.stories.tsx`)
  }
}

if (missing.length > 0) {
  console.error('Story completeness check failed:\n' + missing.map((m) => `  - ${m}`).join('\n'))
  process.exit(1)
}
console.log(`All ${registry.components.length} components have Storybook stories.`)
```

**Goal:** The landing app serves `/`, `/accessibility`, `/performance`, and `/og` from a
dependency-free route map with correct `document.title`s; the header navigates to the new
pages; static deep links work in `dist/`; a typed `virtual:bench` module exposes
sample-stripped bench data to all pages; and `apps/bench/results/results.json` contains real
populated results with provenance.

**Architecture:** `ROUTES` record in `App.tsx` keyed by pathname (extends the `/og` pattern;
plain `<a href>` full-page navigation, no router, no history API — master plan decision 1).
Two stub pages under `apps/landing/src/pages/` (filled in by T3/T4). Two inline Vite plugins
in `apps/landing/vite.config.ts`: one copies `dist/index.html` into per-route folders on
`closeBundle` (decision 2), one resolves `virtual:bench` by reading
`apps/bench/results/results.json` with `node:fs` and deep-deleting every `samples` array
(decision 3). The bench actually runs on this machine (decision 5).

**Tech stack:** unchanged — React 18+ landing with `@preact/signals-react` (`useSignals()`
rule applies), vp (vite+) toolchain. No new dependencies anywhere in this tranche.

- [ ] **Step 3:** Rewrite `apps/landing/src/App.tsx` with the `ROUTES` map. The home
      sections move into a local `HomePage` component (same order as today); the title is
      assigned directly before render — routing is static per page load, so a one-shot
      assignment is correct and avoids any effect machinery (and `useEffect` is banned
      anyway). Trailing slashes are normalized because static hosts serve the deep-link
      copies as `/accessibility/`:

```tsx
import type { ComponentType } from 'react'
import { useSignalEffect } from '@cascade-ui/core'
import { Header } from './sections/Header'
import { Hero } from './sections/Hero'
import { Principles } from './sections/Principles'
import { RelayConsole } from './demo/RelayConsole'
import { SignalsDemo } from './sections/SignalsDemo'
import { AgentLayer } from './sections/AgentLayer'
import { ThemeDemo } from './sections/ThemeDemo'
import { QuickStart } from './sections/QuickStart'
import { CtaBand } from './sections/CtaBand'
import { Footer } from './sections/Footer'
import { OgCard } from './sections/OgCard'
import { AccessibilityPage } from './pages/AccessibilityPage'
import { PerformancePage } from './pages/PerformancePage'
import { initReveal } from './reveal'

function HomePage() {
  return (
    <>
      <Header />
      <Hero />
      <Principles />
      <RelayConsole />
      <SignalsDemo />
      <AgentLayer />
      <ThemeDemo />
      <QuickStart />
      <CtaBand />
      <Footer />
    </>
  )
}

type Route = { Page: ComponentType; title: string }

const HOME: Route = {
  Page: HomePage,
  title: 'cascade — the CSS-native, signal-driven, AI-first React design system',
}

const ROUTES: Record<string, Route> = {
  '/': HOME,
  '/accessibility': { Page: AccessibilityPage, title: 'Accessibility — cascade' },
  '/performance': { Page: PerformancePage, title: 'Performance — cascade' },
  '/og': { Page: OgCard, title: 'cascade' },
}

export function App() {
  useSignalEffect(() => initReveal())

  const pathname =
    typeof window !== 'undefined' ? window.location.pathname.replace(/\/+$/, '') || '/' : '/'
  const route = ROUTES[pathname] ?? HOME

  if (typeof document !== 'undefined') {
    document.title = route.title
  }

  return <route.Page />
}
```

      Notes: `ROUTES[pathname]` is `Route | undefined` under `noUncheckedIndexedAccess`;
      the `?? HOME` fallback (a non-indexed const) keeps it type-clean and makes unknown
      paths render home, same as today. `HOME.title` duplicates `index.html`'s `<title>`
      so a home visit never flashes a different title.

```ts
import { copyFileSync, mkdirSync } from 'node:fs'
import { resolve } from 'node:path'
import { fileURLToPath } from 'node:url'
import { defineConfig, type Plugin } from 'vite-plus'

const __dirname = fileURLToPath(new URL('.', import.meta.url))
const root = resolve(__dirname, '../..')

/** Static hosts have no SPA rewrite rules — copy index.html so deep links resolve. */
const DEEP_LINK_ROUTES = ['accessibility', 'performance']

function deepLinkCopies(): Plugin {
  return {
    name: 'cascade:deep-link-copies',
    apply: 'build',
    closeBundle() {
      const dist = resolve(__dirname, 'dist')
      for (const route of DEEP_LINK_ROUTES) {
        mkdirSync(resolve(dist, route), { recursive: true })
        copyFileSync(resolve(dist, 'index.html'), resolve(dist, route, 'index.html'))
      }
    },
  }
}

export default defineConfig({
  preview: { port: 4180, strictPort: true },
  server: { port: 4180 },
  plugins: [deepLinkCopies()],
  resolve: {
    alias: {
      // Profiling build so the SignalsDemo <Profiler> commit counters work in
      // production bundles (prod react-dom compiles onRender away). Marginally
      // slower than the prod build — intentional on a page that demos re-renders.
      'react-dom/client': 'react-dom/profiling',
      '@cascade-ui/core': resolve(root, 'packages/core/src/index.ts'),
      '@cascade-ui/storage': resolve(root, 'packages/storage/src/index.ts'),
      '@cascade-ui/i18n': resolve(root, 'packages/i18n/src/index.ts'),
      '@cascade-ui/ai': resolve(root, 'packages/ai/src/index.ts'),
      '@cascade-ui/render': resolve(root, 'packages/render/src/index.ts'),
      '@cascade-ui/icons': resolve(root, 'packages/icons/src/index.tsx'),
    },
  },
})
```

```ts
import { copyFileSync, mkdirSync, readFileSync } from 'node:fs'
```

```ts
/** Recursively delete every `samples` array (TimingStats.samples — bundle weight). */
function stripSamples(node: unknown): void {
  if (node === null || typeof node !== 'object') return
  if (Array.isArray(node)) {
    for (const item of node) stripSamples(item)
    return
  }
  const record = node as Record<string, unknown>
  delete record['samples']
  for (const value of Object.values(record)) stripSamples(value)
}

/** `virtual:bench` — trimmed apps/bench/results/results.json as a typed default export. */
function benchData(): Plugin {
  const virtualId = 'virtual:bench'
  const resolvedId = '\0virtual:bench'
  return {
    name: 'cascade:bench-data',
    resolveId(id) {
      return id === virtualId ? resolvedId : null
    },
    load(id) {
      if (id !== resolvedId) return null
      const raw: unknown = JSON.parse(
        readFileSync(resolve(root, 'apps/bench/results/results.json'), 'utf8'),
      )
      stripSamples(raw)
      return `export default ${JSON.stringify(raw)}`
    },
  }
}
```

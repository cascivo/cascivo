- [ ] **Step 8: Add SwapDemo**

  `Swap` takes `on` and `off` slots (ReactNode) and `checked` (boolean). `mode="rotate"` applies a rotation animation between the two states.

  ```tsx
  function SwapDemo() {
    const [checked, setChecked] = useState(false)
    return (
      <Row>
        <Swap
          checked={checked}
          onChange={setChecked}
          on="Light"
          off="Dark"
          mode="rotate"
        />
        <span style={{ fontSize: 'var(--cascivo-text-sm)', color: 'var(--cascivo-color-text-subtle)' }}>
          {checked ? 'Light mode' : 'Dark mode'}
        </span>
      </Row>
    )
  }
  ```

- [ ] **Step 9: Add ToggleGroupDemo**

  `ToggleGroup` takes `type: 'single' | 'multiple'`, `items: ToggleGroupItem[]`, and `value`/`onValueChange`. For `type="single"`, `onValueChange` receives `string`; for `type="multiple"`, `string[]`. Use a type guard.

  ```tsx
  function ToggleGroupDemo() {
    const [value, setValue] = useState<string>('week')
    return (
      <Col>
        <ToggleGroup
          type="single"
          value={value}
          onValueChange={(v) => { if (typeof v === 'string') setValue(v) }}
          items={[
            { value: 'day', label: 'Day' },
            { value: 'week', label: 'Week' },
            { value: 'month', label: 'Month' },
          ]}
        />
      </Col>
    )
  }
  ```

- [ ] **Step 2: Implement `error-boundary.tsx`**

```tsx
/**
 * ErrorBoundary — the one legitimate React class component in cascade.
 *
 * React's error boundary API requires getDerivedStateFromError / componentDidCatch,
 * which are unavailable to function components. This is the documented exception
 * to cascade's no-class-component rule. All styling is handled by the fallback
 * render prop; this component has no CSS of its own.
 */
import { Component, type ComponentType, type ErrorInfo, type ReactNode } from 'react'

export interface ErrorFallbackProps {
  error: Error
  reset: () => void
}

export interface ErrorBoundaryProps {
  children: ReactNode
  fallback: ComponentType<ErrorFallbackProps> | ((props: ErrorFallbackProps) => ReactNode)
  onError?: (error: Error, info: ErrorInfo) => void
  resetKeys?: unknown[]
}

interface State {
  error: Error | null
}

export class ErrorBoundary extends Component<ErrorBoundaryProps, State> {
  constructor(props: ErrorBoundaryProps) {
    super(props)
    this.state = { error: null }
  }

  static getDerivedStateFromError(error: Error): State {
    return { error }
  }

  componentDidCatch(error: Error, info: ErrorInfo): void {
    this.props.onError?.(error, info)
  }

  componentDidUpdate(prevProps: ErrorBoundaryProps): void {
    if (this.state.error === null) return
    const prevKeys = prevProps.resetKeys ?? []
    const nextKeys = this.props.resetKeys ?? []
    if (prevKeys.some((key, i) => key !== nextKeys[i])) {
      this.setState({ error: null })
    }
  }

  reset = (): void => {
    this.setState({ error: null })
  }

  render(): ReactNode {
    const { error } = this.state
    if (error !== null) {
      const FallbackComponent = this.props.fallback
      if (typeof FallbackComponent === 'function') {
        return <FallbackComponent error={error} reset={this.reset} />
      }
    }
    return this.props.children
  }
}
```

```tsx
import { useRef, type ReactNode } from 'react'
import { createPortal } from 'react-dom'
import { useSignal, useSignalEffect } from '@cascade-ui/core'

export interface PortalProps {
  children: ReactNode
  container?: Element | DocumentFragment
}

/**
 * Renders children into document.body (or a custom container) via createPortal.
 * SSR-safe: returns null when document is unavailable.
 */
export function Portal({ children, container }: PortalProps) {
  const mounted = useSignal(false)
  const containerRef = useRef<Element | null>(null)

  useSignalEffect(() => {
    if (typeof document === 'undefined') return
    containerRef.current = container ?? document.body
    mounted.value = true
    return () => { mounted.value = false }
  })

  if (!mounted.value || containerRef.current === null) return null
  return createPortal(children, containerRef.current)
}
```

```tsx
'use client'

import { useRef, type ReactNode } from 'react'
import { useSignalEffect } from '@cascade-ui/core'

const FOCUSABLE =
  'a[href],area[href],input:not([disabled]),select:not([disabled]),textarea:not([disabled]),button:not([disabled]),[tabindex]:not([tabindex="-1"])'

export interface FocusScopeProps {
  children: ReactNode
  /** Trap Tab/Shift-Tab within the scope */
  trapped?: boolean
  /** Restore focus to the previously focused element on unmount */
  restoreFocus?: boolean
  /** Focus the first focusable element on mount */
  autoFocus?: boolean
}

export function FocusScope({ children, trapped, restoreFocus, autoFocus }: FocusScopeProps) {
  const containerRef = useRef<HTMLDivElement>(null)
  const previousFocusRef = useRef<Element | null>(null)

  useSignalEffect(() => {
    const container = containerRef.current
    if (!container) return

    if (restoreFocus) {
      previousFocusRef.current = document.activeElement
    }

    if (autoFocus) {
      const first = container.querySelector<HTMLElement>(FOCUSABLE)
      first?.focus()
    }

    if (!trapped) return

    function handleKeyDown(e: KeyboardEvent) {
      if (e.key !== 'Tab') return
      const focusable = Array.from(container!.querySelectorAll<HTMLElement>(FOCUSABLE))
      if (focusable.length === 0) return
      const first = focusable[0]!
      const last = focusable[focusable.length - 1]!

      if (e.shiftKey) {
        if (document.activeElement === first) {
          e.preventDefault()
          last.focus()
        }
      } else {
        if (document.activeElement === last) {
          e.preventDefault()
          first.focus()
        }
      }
    }

    document.addEventListener('keydown', handleKeyDown)
    return () => {
      document.removeEventListener('keydown', handleKeyDown)
      if (restoreFocus && previousFocusRef.current instanceof HTMLElement) {
        previousFocusRef.current.focus()
      }
    }
  })

  return <div ref={containerRef}>{children}</div>
}
```

```ts
}

const BANNED_HOOKS = ['useState', 'useEffect', 'useLayoutEffect', 'useContext', 'useReducer']

export async function runDoctor({ root }: DoctorOptions): Promise<DoctorResult> {
  const violations: DoctorViolation[] = []
  const componentsDir = join(root, 'packages', 'components', 'src')

  if (!existsSync(componentsDir)) {
    return { violations: [], passed: true }
  }

  const components = readdirSync(componentsDir, { withFileTypes: true })
    .filter(d => d.isDirectory())
    .map(d => d.name)

  for (const name of components) {
    const tsxPath = join(componentsDir, name, `${name}.tsx`)
    if (!existsSync(tsxPath)) continue

    const content = readFileSync(tsxPath, 'utf8')

    // Rule: no-react-hooks — no banned React hooks in component files
    for (const hook of BANNED_HOOKS) {
      if (new RegExp(`\\b${hook}\\b`).test(content)) {
        violations.push({
          file: tsxPath,
          rule: 'no-react-hooks',
          detail: `Banned hook '${hook}' found in ${name}.tsx`,
        })
      }
    }

    // Rule: no-hardcoded-strings — aria-label or text with literal English
    if (/aria-label="[A-Z]/.test(content)) {
      violations.push({
        file: tsxPath,
        rule: 'no-hardcoded-strings',
        detail: `Hardcoded aria-label string found in ${name}.tsx — use t(builtin.*)`,
      })
    }

    // Rule: react-export — component must be exported from @cascade-ui/react
    const reactIndex = join(root, 'packages', 'react', 'src', 'index.ts')
    if (existsSync(reactIndex)) {
      const index = readFileSync(reactIndex, 'utf8')
      const exportPattern = new RegExp(`from.*components/src/${name}`)
      if (!exportPattern.test(index)) {
        violations.push({
          file: tsxPath,
          rule: 'missing-react-export',
          detail: `${name} is not exported from @cascade-ui/react`,
        })
      }
    }
  }

  return { violations, passed: violations.length === 0 }
}
```

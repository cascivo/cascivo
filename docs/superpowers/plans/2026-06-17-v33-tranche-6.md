**Tech Stack:** React, `@cascivo/core` signals, existing `perf-data.ts` selectors, CSS in `landing.css`.

- [ ] **Step 2: Compute best/worst per row before the cell map:**

  ```tsx
  {rows.map((row) => {
    const medians = LIBS.map((lib) => row.stats[lib]?.median).filter(
      (m): m is number => m !== undefined,
    )
    const best = medians.length > 1 ? Math.min(...medians) : undefined
    const worst = medians.length > 1 ? Math.max(...medians) : undefined
    const distinct = best !== undefined && worst !== undefined && best !== worst
    return (
      <tr key={row.scenario}>
        <th scope="row">{SCENARIO_LABELS[row.scenario]}</th>
        {LIBS.map((lib) => {
          const s = row.stats[lib]
          if (!s) return <td key={lib}>‚Äî</td>
          const perf = !distinct
            ? undefined
            : s.median === best
              ? 'best'
              : s.median === worst
                ? 'worst'
                : undefined
          return (
            <td key={lib} data-perf={perf}>
              {fmtMs(s.median)}{' '}
              <span className="perf-iqr">({s.p25.toFixed(1)}‚Äì{s.p75.toFixed(1)})</span>
              {row.ties[lib] ? <sup>‚ÄÝ</sup> : null}
            </td>
          )
        })}
      </tr>
    )
  })}
  ```

  `data-perf` is `undefined` (omitted) when fewer than two libs have data or all medians tie.

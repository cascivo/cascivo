`stack.tsx`:

```tsx
'use client'
import { cn } from '@cascivo/core'
import type { HTMLAttributes } from 'react'
import styles from './stack.module.css'

export interface StackProps extends HTMLAttributes<HTMLDivElement> {
  direction?: 'vertical' | 'horizontal'
  /** Spacing-token step: 1..12 maps to --cascivo-space-N. */
  gap?: 1 | 2 | 3 | 4 | 5 | 6 | 8 | 10 | 12
  align?: 'start' | 'center' | 'end' | 'stretch'
  justify?: 'start' | 'center' | 'end' | 'between'
  wrap?: boolean
}

export function Stack({
  direction = 'vertical',
  gap = 4,
  align,
  justify,
  wrap = false,
  className,
  ...props
}: StackProps) {
  return (
    <div
      className={cn(styles['stack'], className)}
      data-direction={direction}
      data-wrap={wrap || undefined}
      style={{
        ['--_stack-gap' as string]: `var(--cascivo-space-${gap})`,
        ...(align
          ? { alignItems: align === 'start' || align === 'end' ? `flex-${align}` : align }
          : {}),
        ...(justify
          ? {
              justifyContent:
                justify === 'between'
                  ? 'space-between'
                  : justify === 'start' || justify === 'end'
                    ? `flex-${justify}`
                    : justify,
            }
          : {}),
        ...props.style,
      }}
      {...props}
    />
  )
}
```

`stack.module.css`:

```css
@layer cascade.component {
  .stack {
    display: flex;
    gap: var(--_stack-gap, var(--cascivo-space-4));

    &[data-direction='vertical'] {
      flex-direction: column;
    }
    &[data-direction='horizontal'] {
      flex-direction: row;
    }
    &[data-wrap] {
      flex-wrap: wrap;
    }
  }
}
```

`stack.meta.ts`: `category: 'layout'`, dependencies `['@cascivo/core']`, honest props, one example. `stack.test.tsx`: renders children; `data-direction` per prop; gap CSS var set.

### Task 4: AppShell (with persisted sidebar) + SplitView

**Files:** four-file sets under `packages/layouts/src/app-shell/` and `packages/layouts/src/split-view/`.

- [ ] **Step 1: AppShell**

Props:

```tsx
export interface AppShellProps {
  header: ReactNode
  sideNav?: ReactNode
  aside?: ReactNode
  children: ReactNode
  /** storage key for the collapsed state; set to disable persistence with `false` */
  persistKey?: string | false
  className?: string
}
```

Behavior: CSS grid `grid-template: 'header header header' auto 'nav main aside' 1fr / auto minmax(0, 1fr) auto`. Collapse state is a signal; when `persistKey` is a string (default `'cascade.appshell.sidenav'`), use `persistedSignal(persistKey, false)` — created lazily per mount with the same `useSignal`-lazy-init pattern as `useForm` (tranche 2 Task 10):

```tsx
const store = useSignal<Signal<boolean> | null>(null)
if (store.peek() === null) {
  store.value = persistKey === false ? signal(false) : persistedSignal(persistKey, false)
}
const collapsed = store.peek() as Signal<boolean>
```

A collapse toggle button (aria-label from a new `builtin.appShell` namespace: `collapse`/`expand`, en+de — extend `packages/i18n/src/builtin.ts`) sets `collapsed.value`. The nav column gets `data-state={collapsed.value ? 'collapsed' : 'expanded'}`; CSS transitions only `translate`/`opacity` on the nav content (width snap is instant — animating width is forbidden by the audit gate; the rail pattern: collapsed shows icon rail via `inline-size` swap without transition).

Container-query responsive: below `48rem` the nav overlays (position absolute, full-height, `@starting-style` slide-in) instead of occupying a column.

Tests: renders slots; toggle flips `data-state`; persistence — set, unmount, remount with same key → still collapsed (localStorage envelope present).

- [ ] **Step 2: SplitView**

Props: `start: ReactNode`, `end: ReactNode`, `defaultRatio?: number` (0.3), `min?: number` (0.2), `max?: number` (0.8), `persistKey?: string | false` (default false), `aria-label` for the separator.

Drag: ratio is a signal; the divider is `role="separator"` with `aria-orientation="vertical"`, `tabIndex={0}`, `aria-valuenow` = percent. Pointer events on the divider: `onPointerDown` captures (`setPointerCapture`), `onPointerMove` computes ratio from `getBoundingClientRect` of the root and writes the signal; a `useSignalEffect` writes `--_split-ratio` onto the root element style via a ref (zero React re-render drags). Keyboard: ArrowLeft/Right adjust by 0.02 within min/max. CSS: `grid-template-columns: minmax(0, calc(var(--_split-ratio) * 100%)) auto minmax(0, 1fr)`.

Tests: renders panes; keyboard arrows change `aria-valuenow`; ratio clamped to min/max.

```tsx
import { Badge, Button, type Column, DataTable } from '@cascivo/react'
import { PageHeader } from '../../page-header/page-header'
import { Stack } from '../../stack/stack'

interface User {
  id: string
  name: string
  email: string
  role: 'admin' | 'member'
  active: boolean
}

const demoUsers: User[] = [
  { id: '1', name: 'Ada Lovelace', email: 'ada@example.com', role: 'admin', active: true },
  { id: '2', name: 'Grace Hopper', email: 'grace@example.com', role: 'member', active: true },
  { id: '3', name: 'Alan Turing', email: 'alan@example.com', role: 'member', active: false },
]

const columns: Column<User>[] = [
  { key: 'name', header: 'Name', sortable: true },
  { key: 'email', header: 'Email' },
  {
    key: 'role',
    header: 'Role',
    render: (user) => (
      <Badge variant={user.role === 'admin' ? 'accent' : 'neutral'}>{user.role}</Badge>
    ),
  },
]

export interface UsersTablePageProps {
  users?: User[]
  onInvite?: () => void
}

export function UsersTablePage({ users = demoUsers, onInvite }: UsersTablePageProps) {
  return (
    <Stack gap={6}>
      <PageHeader
        title="Users"
        description="Manage who has access to this workspace."
        actions={<Button onClick={onInvite}>Invite user</Button>}
      />
      <DataTable
        columns={columns}
        rows={users}
        getRowId={(u) => u.id}
        searchable
        pagination={{ pageSize: 10 }}
        selection={{ mode: 'multi' }}
      />
    </Stack>
  )
}
```

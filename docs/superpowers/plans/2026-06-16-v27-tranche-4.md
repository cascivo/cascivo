# v27 — Tranche 4: Quick Search — `@cascivo/search` + CMD+K Dialog

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create the `@cascivo/search` workspace package (SearchIndex + SearchDialog), wire it
into the landing app, and expose CMD+K search to visitors. The index covers all registry
components and the landing's key page sections.

**Architecture:**
- `packages/search/` — workspace package, source only (not published in v27). Exports
  `SearchIndex` (pure TS class, no React) and `SearchDialog` (React component using
  `@cascivo/components` primitives).
- `apps/landing/src/search/` — landing-specific: index builder, search icon button.
- `App.tsx` — mounts the CMD+K keyboard handler and renders `<SearchDialog>`.
- `Header.tsx` — adds a search icon button that opens the dialog.

**Tech Stack:** `@cascivo/core` (useSignal, useSignals, useSignalEffect), `@cascivo/components`
(Modal or Dialog, Input, Kbd), registry.json, plain CSS.

**Prerequisite:** T1 (SPA router) must be complete — `navigate()` is used from `router.ts` to
navigate to search results.

- [ ] **Step 2: Create `packages/search/src/index.ts`**

  ```ts
  export interface SearchItem {
    id: string
    title: string
    section?: string
    description?: string
    href: string
    type: 'component' | 'page'
    category?: string
  }

  export class SearchIndex {
    private items: SearchItem[]

    constructor(items: SearchItem[]) {
      this.items = items
    }

    search(query: string, limit = 8): SearchItem[] {
      const q = query.toLowerCase().trim()
      if (q.length < 2) return []

      const score = (item: SearchItem): number => {
        const title = item.title.toLowerCase()
        const desc = item.description?.toLowerCase() ?? ''
        const section = item.section?.toLowerCase() ?? ''
        const category = item.category?.toLowerCase() ?? ''

        if (title === q) return 100
        if (title.startsWith(q)) return 80
        if (title.includes(q)) return 60
        if (desc.includes(q)) return 40
        if (section.includes(q) || category.includes(q)) return 20
        return 0
      }

      return this.items
        .map((item) => ({ item, score: score(item) }))
        .filter(({ score }) => score > 0)
        .sort((a, b) => b.score - a.score)
        .slice(0, limit)
        .map(({ item }) => item)
    }
  }
  ```

- [ ] **Step 3: Create `packages/search/src/SearchDialog.tsx`**

  This component requires reading the available Modal/Dialog and Input components first (Task 1).
  The implementation follows this structure:

  ```tsx
  'use client'
  import { useComputed, useSignal, useSignalEffect, useSignals } from '@cascivo/core'
  import { Modal } from '@cascivo/components/modal'  // adjust path based on audit
  import { Input } from '@cascivo/components/input'
  import type { SearchIndex, SearchItem } from './index'

  export interface SearchDialogProps {
    index: SearchIndex
    open: boolean
    onClose: () => void
    onNavigate: (href: string) => void
  }

  export function SearchDialog({ index, open, onClose, onNavigate }: SearchDialogProps) {
    useSignals()
    const query = useSignal('')
    const cursor = useSignal(0)
    const results = useComputed<SearchItem[]>(() => index.search(query.value))

    // Reset on open
    useSignalEffect(() => {
      if (open) {
        query.value = ''
        cursor.value = 0
      }
    })

    // Keyboard navigation within the list
    useSignalEffect(() => {
      if (!open) return
      const handler = (e: KeyboardEvent) => {
        if (e.key === 'ArrowDown') {
          e.preventDefault()
          cursor.value = Math.min(cursor.value + 1, results.value.length - 1)
        } else if (e.key === 'ArrowUp') {
          e.preventDefault()
          cursor.value = Math.max(cursor.value - 1, 0)
        } else if (e.key === 'Enter') {
          const item = results.value[cursor.value]
          if (item) {
            onNavigate(item.href)
            onClose()
          }
        }
      }
      document.addEventListener('keydown', handler)
      return () => document.removeEventListener('keydown', handler)
    })

    return (
      <Modal
        open={open}
        onClose={onClose}
        aria-label="Search"
        className="search-dialog"
      >
        <div className="search-dialog-inner">
          <Input
            autoFocus
            placeholder="Search components, guides…"
            value={query.value}
            onChange={(e) => {
              query.value = e.target.value
              cursor.value = 0
            }}
            className="search-input"
          />
          {results.value.length > 0 && (
            <ul className="search-results" role="listbox">
              {results.value.map((item, i) => (
                <li
                  key={item.id}
                  role="option"
                  aria-selected={i === cursor.value}
                  className={`search-result${i === cursor.value ? ' search-result--active' : ''}`}
                  onMouseEnter={() => { cursor.value = i }}
                  onClick={() => {
                    onNavigate(item.href)
                    onClose()
                  }}
                >
                  <span className="search-result-title">{item.title}</span>
                  {item.section && (
                    <span className="search-result-section">{item.section}</span>
                  )}
                  {item.description && (
                    <span className="search-result-desc">{item.description}</span>
                  )}
                  <span className={`search-result-type search-result-type--${item.type}`}>
                    {item.category ?? item.type}
                  </span>
                </li>
              ))}
            </ul>
          )}
          {query.value.length >= 2 && results.value.length === 0 && (
            <p className="search-empty">No results for "{query.value}"</p>
          )}
        </div>
      </Modal>
    )
  }
  ```

  Adjust `Modal` import path and props to match what the audit in Task 1 reveals.

### Task 5: Wire `SearchDialog` into `App.tsx`

- [ ] **Step 1: Lift `searchOpen` signal to `App`**

  In `App.tsx`:

  ```tsx
  import { SearchDialog } from '@cascivo/search/SearchDialog'
  import { landingIndex } from './search/buildIndex'
  import { navigate } from './router'

  export function App() {
    useSignals()
    const searchOpen = useSignal(false)

    // CMD+K / Ctrl+K handler
    useSignalEffect(() => {
      const handler = (e: KeyboardEvent) => {
        if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
          e.preventDefault()
          searchOpen.value = true
        }
      }
      document.addEventListener('keydown', handler)
      return () => document.removeEventListener('keydown', handler)
    })

    // ... existing router + reveal effects ...

    return (
      <>
        <Suspense ...>
          <route.Page searchOpen={searchOpen} />
        </Suspense>
        <SearchDialog
          index={landingIndex}
          open={searchOpen.value}
          onClose={() => { searchOpen.value = false }}
          onNavigate={(href) => navigate(href)}
        />
      </>
    )
  }
  ```

  The `searchOpen` signal needs to be passed to `Header` so the `SearchButton` can open it.
  Pass it as a prop to `HomePage` → `Header`, or lift it to module scope. The cleanest approach
  for signals: make `searchOpen` a module-level signal in a shared file (e.g.,
  `apps/landing/src/search/state.ts`) and import it in both `App.tsx` and `Header.tsx`.
